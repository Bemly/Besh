43// rust 有各种平台实现，可以跨平台编译使用
44pub fn main(args: Vec<String>) -> Result<(), Box<dyn std::error::Error>> {
45
46    println!("Welcome to the Bemly shell!\n");
47
48    // 快速执行命令
49    if args.len() != 0 {
50
51        // match cmd.get(0).unwrap().to_lowercase().trim() {
52        //     // 指定参数功能输出
53        //     "-h" | "--help" => println!("toggle use libc shell: -u | --use-libc"),
54        //     _ => todo!()
55        // }
56
57        // exec_cmd：命令执行体，命令名
58        // 命令执行体：命令名，参数数组
59        let cmd = args.get(0).unwrap().as_str();
60        exec_cmd(Command::new(cmd).args(args.iter().skip(1)).output(), cmd)?;
61    }
62
63    // 获取用户名和主机名 第一层环境变量 第二层使用软件 大气层恐慌
64    let username = env::var("USER")
65        .unwrap_or(String::from_utf8(Command::new("/usr/bin/whoami").output()?.stdout)?);
66    let hostname = fs::read_to_string("/etc/hostname")
67        .unwrap_or(String::from("unknown")).replace("\n", "");
68    
69    // 获取当前工作目录 第一层环境变量 第二层使用软件 大气层恐慌
70    let mut pwd = env::current_dir()
71        .unwrap_or(
72            fs::canonicalize(
73                String::from_utf8(Command::new("/usr/bin/pwd").output()?.stdout)?.replace("\n", "")
74            )?
75        );
76
77    let homedir = Homedir::init(env::var("HOME").unwrap_or(String::new()));
78
79    // 进入循环执行模式
80    loop {
81        // 显示：获取用户目录
82        print!("{username}@{hostname} {}> ", homedir.to_relative_home(&pwd));
83        stdout().flush()?;
84        // 读取用户输入
85        let mut command_buffer = String::new();
86        stdin().read_line(&mut command_buffer)?;
87
88        // 捕获指定命令
89        // TODO: 屎山代码 分割空格为 实现了迭代器方法的spw对象
90        // cmd从program弹出，剩下的是参数
91        let mut program = command_buffer.trim().split_whitespace();
92        let cmd = program.next().unwrap_or("");
93        match cmd {
94            // 退出程序 最穷举的一集
95            "exit" | "exit()" | "quit" | "qui" | "qu" | "q" | ":q" => exit(0),
96            // 就是空行 上方None替换
97            "" => println!(),
98            "cd" => {
99                // 没有参数
100                let args = program.next().unwrap_or("");
101                // 多个参数
102                if args.eq("") { println!() }
103                else if program.count().eq(&0usize) {
104                    homedir.to_absoulte_home(args, &mut pwd);
105                    println!("{pwd:?}");
106                    // unimplemented!  替换~的Bug 暂时不想修了 做libc调用的时候再考虑
107                    pwd = match fs::canonicalize(&pwd) {
108                        Ok(pwd) => {
109                            println!("{cmd}: success change dir to : {args} 成功切换目录到: {args}");
110                            pwd
111                        },
112                        Err(e) => {
113                            println!("besh: {cmd}: {e}");
114                            PathBuf::from(homedir.path.clone())
115                        }
116                    }
117                    
118                } else {
119                    println!("besh: {cmd}: too many arguments. 只允许一个参数")
120                }
121            },
122            // 执行路径下软件或者脚本
123            _ => exec_cmd(Command::new(cmd).args(program).current_dir(&pwd).output(), cmd)?
124        }
125    }
126}

128// 执行命令
129fn exec_cmd(result: Result<Output, io::Error>, cmd: &str) -> Result<(), Box<dyn std::error::Error>> {
130    match result {
131        // 匹配Result()结果，成功则从stdout流打印输出结果，失败打印stderr流
132        Ok(output) => { println!("{}", String::from_utf8(output.stdout)?) },
133        Err(e) => { eprintln!("besh: {cmd}: {e}") }
134    }
135    Ok(())
136}

138struct Homedir<'a> {
139    path: String,
140    name: &'a str
141}
142impl<'a> Homedir<'a> {
143    fn init(path: String) -> Self {
144        Homedir { path, name: "~" }
145    }
146    fn to_absoulte_home(&self, args: &str, pwd: &mut PathBuf) {
147        // 替换掉~通配符之后再push拼接
148        if args.find(self.name).unwrap_or(0).eq(&0) {
149            pwd.push(args.replacen(self.name, &self.path, 1))
150        }
151    }
152
153    fn to_relative_home(&self, pwd: &PathBuf) -> String {
154        pwd.to_str().unwrap_or("").replacen(&self.path, self.name, 1)
155    }
156}
